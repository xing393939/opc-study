$$
\tag{向量与常量的乘法}
\begin{vmatrix}
1\\
9
\end{vmatrix}
*c
=
\begin{vmatrix}
1*c\\
9*c
\end{vmatrix}
$$

$$
\tag{向量与向量的加法}
\begin{vmatrix}
1\\
1
\end{vmatrix}
+
\begin{vmatrix}
2\\
3
\end{vmatrix}
=
\begin{vmatrix}
1+2\\
1+3
\end{vmatrix}
$$
